Campaign
-
cf_id INTEGER(5) PK
contact_id INTEGER(5) FK >- Contacts.contact_id
company_name VARCHAR(50)
description VARCHAR(150)
goal INTEGER(10)
pledged INTEGER(10)
outcome VARCHAR(10)
backers_count INTEGER(5)
country VARCHAR(2)
currency VARCHAR(3)
launched_date DATE(10)
end_date DATE(10)
category_id VARCHAR(10) FK >- Category.category_id
subcategory_id VARCHAR(10) FK >- Subcategory.subcategory_id
last_updated TIMESTAMP


Category
-
category_id VARCHAR(5) PK
category VARCHAR(20)
last_updated TIMESTAMP


Contacts
-
contact_id INTEGER(5) PK
first_name VARCHAR(20)
last_name VARCHAR(20)
email VARCHAR(50)
last_updated TIMESTAMP

Subcategory
-
subcategory_id VARCHAR(10) PK
subcategory VARCHAR(30)
